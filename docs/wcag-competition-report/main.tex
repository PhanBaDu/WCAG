\section{Giới thiệu dự án}
\projectname{} là một website hỗ trợ tìm việc và tuyển dụng tập trung cho người khuyết tật, với mục tiêu cung cấp trải nghiệm tiếp cận rõ ràng, dễ điều hướng và phù hợp với các nguyên tắc WCAG 2.2.

Phiên bản hiện tại của sản phẩm được xây dựng theo hướng giao diện tĩnh để chốt bố cục, thứ bậc nội dung, tương phản, hành vi bàn phím và tính nhất quán của các thành phần. Điều này giúp đội thi tập trung chứng minh năng lực tiếp cận trước, sau đó mới mở rộng sang dữ liệu động.

\section{Tài liệu này dành cho ai}
Tài liệu báo cáo này được soạn chủ yếu cho \textbf{ban giám khảo, hội đồng kỹ thuật và ban tổ chức cuộc thi}. Khi đọc tài liệu, nhóm người này thường cần trả lời bốn câu hỏi:
\begin{itemize}
  \item Sản phẩm này \textbf{giải quyết vấn đề gì} và có thật sự hữu ích không?
  \item Sản phẩm \textbf{phục vụ ai} và ai là người dùng chính?
  \item Sản phẩm \textbf{vận hành theo những actor nào} và actor nào là trọng tâm?
  \item Sản phẩm đã \textbf{áp dụng WCAG 2.2} như thế nào, có kiểm thử thực tế chưa?
\end{itemize}

Vì vậy, báo cáo không chỉ mô tả giao diện, mà phải mô tả được \textbf{bối cảnh nghiệp vụ}, \textbf{actor}, \textbf{use case}, \textbf{luồng thao tác}, và \textbf{tiêu chí tiếp cận} của một hệ thống hoàn chỉnh.

\section{Bài toán và giá trị}
\begin{itemize}
  \item \textbf{Bài toán}: người khuyết tật thường gặp khó khăn khi tìm việc vì giao diện thiếu tiếp cận, thông tin rời rạc hoặc thao tác bàn phím không ổn định.
  \item \textbf{Giá trị}: \projectname{} tạo một cổng việc làm và tuyển dụng thân thiện, dễ đọc, dễ tab, dễ hiểu và có song ngữ.
  \item \textbf{Định hướng}: không làm “trang web cho đẹp”, mà làm một \textbf{hệ thống có nghiệp vụ rõ}, có actor rõ, có quy trình rõ, và có tiêu chuẩn tiếp cận rõ.
\end{itemize}

\section{Mục tiêu và phạm vi}
\subsection{Mục tiêu}
\begin{itemize}
  \item Tạo một giao diện tìm việc và đăng tin tuyển dụng dễ dùng với bàn phím.
  \item Đảm bảo cấu trúc thông tin rõ ràng cho màn hình đọc và người dùng phóng to giao diện.
  \item Áp dụng cách trình bày nội dung, màu sắc và hình ảnh phù hợp WCAG 2.2 Level AA.
  \item Hỗ trợ song ngữ Việt/Anh để phù hợp với sản phẩm thi và khả năng mở rộng.
\end{itemize}

\subsection{Phạm vi}
\begin{itemize}
  \item Trang chủ giới thiệu sản phẩm.
  \item Trang danh sách việc làm, trang chi tiết việc làm.
  \item Trang đăng nhập, đăng ký, quên mật khẩu, đặt lại mật khẩu.
  \item Hồ sơ ứng viên, dashboard nhà tuyển dụng, form đăng tin.
  \item Tài liệu và hướng dẫn bàn phím phục vụ kiểm tra tiếp cận.
\end{itemize}

\subsection{Use case chi tiết}
\subsubsection{Người tìm việc khuyết tật}
\begin{enumerate}
  \item Truy cập trang chủ và dùng skip link hoặc bàn phím để vào nội dung chính.
  \item Tìm kiếm việc làm theo từ khóa, kỹ năng hoặc địa điểm.
  \item Lọc việc theo dạng khuyết tật, loại công việc và khu vực.
  \item Mở trang chi tiết để xem mô tả, quyền lợi và hỗ trợ tiếp cận.
  \item Ứng tuyển bằng form có label rõ và cấu trúc dễ điền.
  \item Cập nhật hồ sơ cá nhân, kỹ năng, hỗ trợ cần thiết và CV.
\end{enumerate}

\subsubsection{Nhà tuyển dụng}
\begin{enumerate}
  \item Đăng nhập vào khu vực nhà tuyển dụng.
  \item Xem dashboard tổng quan số lượng tin, số lượng ứng viên, tin chờ duyệt.
  \item Tạo tin mới bằng form đăng tin.
  \item Lưu nháp hoặc gửi tin chờ duyệt.
  \item Quản lý các tin đã đăng: sửa, đóng, xóa.
  \item Theo dõi danh sách việc làm và trạng thái hiển thị.
\end{enumerate}

\subsubsection{Quản trị hệ thống}
\begin{enumerate}
  \item Kiểm tra dữ liệu hiển thị và tính đúng của nội dung.
  \item Đảm bảo cấu trúc ngôn ngữ, landmark, tiêu đề và điều hướng nhất quán.
  \item Xử lý các tình huống duyệt nội dung hoặc gỡ nội dung nếu cần.
  \item Hỗ trợ ban tổ chức khi cần rà soát chất lượng sản phẩm.
\end{enumerate}

\section{Đối tượng sử dụng}
\begin{itemize}
  \item \textbf{Người tìm việc khuyết tật}: cần đọc rõ ràng, focus dễ nhận biết, form dễ điền và điều hướng bàn phím mạch lạc.
  \item \textbf{Nhà tuyển dụng}: cần giao diện đăng tin, quản lý tin và xem thống kê trực quan nhưng không gây quá tải thông tin.
  \item \textbf{Người dùng màn hình đọc}: cần landmark đúng, heading có thứ bậc, label rõ ràng và cấu trúc tuần tự.
  \item \textbf{Người dùng bàn phím}: cần skip link, focus visible, tab order hợp lý và không bị focus thoát ra ngoài ứng dụng.
\end{itemize}

\section{Giải pháp tiếp cận theo WCAG 2.2}
\subsection{Cấu trúc và ngữ nghĩa}
\begin{itemize}
  \item Dùng landmark chuẩn: \texttt{header}, \texttt{nav}, \texttt{main}, \texttt{aside}, \texttt{footer}.
  \item Dùng heading theo cấp bậc rõ ràng từ \texttt{h1} đến \texttt{h3}.
  \item Form được tổ chức bằng \texttt{label}, \texttt{fieldset}, \texttt{legend}, bảng và nút có ý nghĩa rõ.
  \item Nội dung động được thay bằng content tĩnh có mô tả đầy đủ để kiểm tra layout và thứ bậc.
\end{itemize}

\subsection{Điều hướng bàn phím}
\begin{itemize}
  \item Có skip link để bỏ qua khối điều hướng lặp lại.
  \item Tăng độ nhận diện focus bằng highlight và/hoặc outline rõ ràng.
  \item Giữ focus trong phạm vi ứng dụng, tránh thoát sang chrome UI ngoài ý muốn.
  \item Bổ sung phần demo shortcut để người dùng hiểu cách di chuyển qua heading, link, button, form field, table và landmark.
\end{itemize}

\subsection{Hình ảnh và media}
\begin{itemize}
  \item Chỉ dùng ảnh có license rõ ràng hoặc ảnh tự tạo cục bộ.
  \item Ảnh có ý nghĩa phải có \texttt{alt} mô tả đúng ngữ cảnh.
  \item Ảnh trang trí phải ẩn khỏi màn hình đọc bằng \texttt{alt=""} nếu không mang thông tin.
  \item Tránh dùng ảnh chứa chữ quan trọng, thay bằng text thật trong DOM.
\end{itemize}

\subsection{Màu sắc và tương phản}
\begin{itemize}
  \item Tập trung vào nền sáng, chữ đậm, khoảng trắng rõ ràng.
  \item Không dùng màu làm kênh duy nhất để truyền tải ý nghĩa.
  \item Focus highlight được tách riêng cho link và control quan trọng theo tinh thần GOV.UK.
  \item Kiểm tra lại tương phản ở cả light mode và dark mode.
\end{itemize}

\subsection{Responsive và reflow}
\begin{itemize}
  \item Kiểm tra ở các viewport nhỏ như 320px và 375px.
  \item Đảm bảo nội dung vẫn đọc được khi phóng to và không mất chức năng.
  \item Hạn chế layout phụ thuộc chiều ngang quá lớn, đặc biệt ở dashboard và trang job listing.
\end{itemize}

\subsection{Đa ngôn ngữ}
\begin{itemize}
  \item Hỗ trợ tiếng Việt và tiếng Anh bằng \texttt{next-intl}.
  \item Gắn \texttt{lang} đúng trên \texttt{html}.
  \item Dịch cả navigation, CTA, mô tả và nhãn form để người dùng hiểu toàn bộ hành vi giao diện.
\end{itemize}

\section{Kiểm thử và xác nhận}
\subsection{Kiểm thử thủ công}
\begin{itemize}
  \item Dùng Tab, Shift+Tab, Enter, Space và Esc để kiểm tra điều hướng.
  \item Kiểm tra heading, link, button, form field, table, landmark.
  \item Kiểm tra ở mobile viewport và khi phóng to giao diện.
  \item Đảm bảo focus không nhảy ra UI của trình duyệt.
\end{itemize}

\subsection{Kiểm thử build}
\begin{itemize}
  \item Chạy \texttt{npm run build} để xác nhận TypeScript và Next.js đều biên dịch tốt.

\end{itemize}
